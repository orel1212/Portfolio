def get_identical_bytes(key1, key2):
    return (numpy.array(list(key1), numpy.uint8) == numpy.array(list(key2), numpy.uint8)).sum()
#

key_hex = ex02_M2.recover_key(TRACES_FILE_NAME, show_plots=True)
key = codecs.decode(key_hex, 'hex')
if not ex02_M2.check_key(key, ex02_M2.USER, ex02_M2.DIFFICULTY):
    raise ValueError('Invalid key :(')

    LOG.debug(f'FINISH {trace_count} traces => {correct_bytes} correct bytes')
